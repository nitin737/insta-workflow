% Custom commands
\newcommand{\resumeItem}[1]{
  \item\small{
    { #1 \vspace{-2pt}}
  }
}

\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{ #1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeProjectHeading}[2]{
    \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubItem}[1]{\resumeItem{ #1}\vspace{-4pt}}

\renewcommand{\labelitemii}{$\vcenter{\hbox{\tiny$\bullet$}}$}

\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=0.15in, label={}]}
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}
\newcommand{\resumeItemListStart}{\begin{itemize}}
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-5pt}}

    [# th:if="${basics['github']}"] $|$ \href{https://[(${latexUtils.escape(basics['github'])})]}{ [(${latexUtils.escape(basics['github'])})] } [/]
\end{center}

%----------PROFESSIONAL SUMMARY----------
\section{Professional Summary}

    \small{\item{
    [(${latexUtils.escape(basics['summary'])})]
    }}

%-----------TECHNICAL SKILLS-----------
[# th:if="${skills}"]
\section{Technical Skills}
\begin{itemize}[leftmargin=0.15in, label={}]
    \small{\item{
    [# th:each="entry : ${skills}"]
    \textbf{ [(${latexUtils.escape(entry.key)})]:} [(${latexUtils.escape(entry.value)})] [# th:if="${!entryStat.last}"]\\ [/]
    [/]
    }}
\end{itemize}
[/]

%-----------PROFESSIONAL EXPERIENCE-----------
\section{Professional Experience}
  \resumeSubHeadingListStart
    [# th:each="exp : ${experience}"]
    \resumeSubheading
      { [(${latexUtils.escape(exp['company'])})] }{ [(${latexUtils.escape(exp['location'])})] }
      { [(${latexUtils.escape(exp['title'])})] }{ [(${latexUtils.escape(exp['date'])})] }
      \resumeItemListStart
        [# th:each="hl : ${exp['highlights']}"]
        \resumeItem{ [(${latexUtils.escape(hl)})] }
        [/]
      \resumeItemListEnd
    [/]
  \resumeSubHeadingListEnd

%-----------EDUCATION-----------
\section{Education}
  \resumeSubHeadingListStart
    [# th:each="edu : ${education}"]
    \resumeSubheading
      { [(${latexUtils.escape(edu['institution'])})] }{ [(${latexUtils.escape(edu['location'])})] }
      { [(${latexUtils.escape(edu['degree'])})] }{ [(${latexUtils.escape(edu['date'])})] }
    [/]
  \resumeSubHeadingListEnd

%-----------CERTIFICATIONS-----------
[# th:if="${certifications}"]
\section{Certifications}
  \resumeSubHeadingListStart
    [# th:each="cert : ${certifications}"]

      {\textbf{ [(${latexUtils.escape(cert['name'])})] } $|$ \emph{ [(${latexUtils.escape(cert['issuer'])})] }}{ [(${latexUtils.escape(cert['date'])})] }
    [/]
  \resumeSubHeadingListEnd
[/]
